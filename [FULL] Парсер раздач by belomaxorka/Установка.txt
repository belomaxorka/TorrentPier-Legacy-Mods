--------------------- Открыть library/config.php (В конце вставить) ---------------------
// ----------------------------------------------------------------
// Парсер раздач
// ----------------------------------------------------------------
// Сделать парсер доступным для всех желающих (Для авторизованных)
$bb_cfg['parser__for_all'] = false;
// ----------------------------------------------------------------
// Тайм-аут для подключения к источнику (По умолчанию 10 секунд)
$bb_cfg['parser__connect_timeout'] = 10;
// ----------------------------------------------------------------
// Фейковый user-agent от лица которого будут парситься раздачи
// Если значение стоит 'random', то будет генерироваться случайный user-agent (Рекомендуется)
// Так же доступны режимы: firefox, chrome, mobile, windows, mac, iphone, ipad и android
// Можно использовать обычные user-agent строчки (Не рекомендуется)
$bb_cfg['parser__user_agent'] = 'random';
// ----------------------------------------------------------------
// Путь до файлы в который будут записываться / читаться куки
$bb_cfg['parser__cookie_path'] = INT_DATA_DIR . '/parser_cookies.txt';
